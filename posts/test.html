Let's make a Minecraft server in Haskell

Posted on 03-01-2016 
import Data.Conduit

main :: IO ()
main = do
  runTCPServer (serverSettings 25566 "*") $ \incoming -> do
    packetSource incoming
    $$ deserializeStatus
    =$= handleStatus
    =$= serializeStatus
    =$= packetSink incoming
